### Single Lesson Video Workflow

```yaml
workflow_single_lesson:
  input:
    - Lesson topic and objectives
    - Key concepts to cover
    - Duration target (5-10 minutes)

  step_1_script:
    tool: Claude / GPT-4
    output: Structured script with timing markers
    template: |
      HOOK (0:00-0:30): Attention-grabbing opening

    tool: Descript / Premiere Pro
    tasks:
      - Assemble all clips

    settings:
      format: MP4 (H.264)
      resolution: 1920x1080
      bitrate: 15 Mbps
```

## Lip Sync Quality Standards

### The SYNC Framework

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                         SYNC QUALITY FRAMEWORK                               │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│  S — SYNCHRONIZATION                                                         │
│  ├── Audio-visual alignment within 50ms                                     │
│  ├── Phoneme-to-viseme mapping accuracy >95%                                │
│  ├── No drift over extended clips (>60 seconds)                             │
│  └── Consistent sync across scene cuts                                      │
│                                                                              │
│  Y — YAWN TEST (Naturalness)                                                 │
│  ├── Jaw movement matches speech intensity                                  │
│  ├── Natural blink rate (15-20 per minute)                                  │
│  ├── Micro-expressions present                                              │
│  └── Head movement complements speech                                       │
│                                                                              │
│  N — NO ARTIFACTS                                                            │
│  ├── No visible warping or distortion                                       │
│  ├── Clean lip boundaries                                                   │
│  ├── Consistent skin tone                                                   │
│  └── No temporal flickering                                                 │
│                                                                              │
│  C — CONSISTENCY                                                             │
│  ├── Same quality across all course videos                                  │
│  ├── Matching style and presentation                                        │
│  ├── Uniform audio levels                                                   │
│  └── Consistent branding elements                                           │
│                                                                              │
│  QUALITY CHECKLIST:                                                          │
│  □ Watch at 0.5x speed for sync issues                                      │
│  □ Check first/last word alignment                                          │
│  □ Verify bilabial sounds (P, B, M)                                         │
│  □ Test labiodental sounds (F, V)                                           │
│  □ Confirm open vowels (A, O)                                               │
│  □ Review scene transitions                                                 │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```

### Lip Sync Troubleshooting

| Issue | Cause | Solution |
|-------|-------|----------|
| Delayed lip movement | Audio leads video | Offset audio by -50 to -100ms |
| Early lip movement | Video leads audio | Offset audio by +50 to +100ms |
| Robotic movement | Low quality model | Use higher-tier avatar or re-render |
| Drift over time | Processing error | Split into shorter clips, rejoin |
| Missing consonants | Fast speech | Add pauses in script, reduce speed |
| Frozen expression | Static avatar | Use Studio Avatar with expressions |

```typescript
// Video upload to Supabase Storage
const uploadCourseVideo = async (
  file: File,
  courseId: string,
  lessonId: string
) => {
  const path = `courses/${courseId}/lessons/${lessonId}/video.mp4`;

  const { data, error } = await supabase.storage
    .from('course-videos')
    .upload(path, file, {
      cacheControl: '3600',
      upsert: true,
      contentType: 'video/mp4'
    });

  // Update lesson record with video URL
  await supabase
    .from('lessons')
    .update({
      video_url: data.path,
      duration: videoDuration,
      has_captions: true
    })
    .eq('id', lessonId);
};
```
